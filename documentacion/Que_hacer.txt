// GET /api/produccion — Listar todas las producciones
if (preg_match('#^/api/produccion/?$#', $path) && $method === 'GET') {
    $animalId = $_GET['animal'] ?? null;
    
    if ($animalId) {
        $sql = "SELECT p.*, a.nombre as nombre_animal, a.codigo_identificacion as codigo
                FROM PRODUCCION_LECHE p
                JOIN ANIMAL a ON p.id_animal = a.id_animal
                WHERE p.id_animal = :id_animal
                ORDER BY p.fecha_produccion DESC";
        $stmt = oci_parse($conn, $sql);
        oci_bind_by_name($stmt, ':id_animal', $animalId);
    } else {
        $sql = "SELECT p.*, a.nombre as nombre_animal, a.codigo_identificacion as codigo
                FROM PRODUCCION_LECHE p
                JOIN ANIMAL a ON p.id_animal = a.id_animal
                ORDER BY p.fecha_produccion DESC";
        $stmt = oci_parse($conn, $sql);
    }
    
    oci_execute($stmt);
    $registros = [];
    while ($row = oci_fetch_assoc($stmt)) {
        $registros[] = [
            'id'               => (int)$row['ID_PRODUCCION'],
            'idAnimal'         => (int)$row['ID_ANIMAL'],
            'nombreAnimal'     => $row['NOMBRE_ANIMAL'],
            'codigoAnimal'     => $row['CODIGO'],
            'fechaProduccion'  => $row['FECHA_PRODUCCION'],
            'litros'           => (float)$row['LITROS'],
            'turno'            => $row['TURNO'],
            'observaciones'    => $row['OBSERVACIONES'],
        ];
    }
    oci_free_statement($stmt);
    http_response_code(200);
    echo json_encode(['data' => $registros], JSON_UNESCAPED_UNICODE);
    exit;
}

// POST /api/produccion — Registrar producción
if (preg_match('#^/api/produccion/?$#', $path) && $method === 'POST') {
    $idAnimal        = $data['idAnimal'] ?? null;
    $fechaProduccion = $data['fechaProduccion'] ?? date('Y-m-d');
    $litros          = $data['litros'] ?? null;
    $turno           = $data['turno'] ?? 'manana';
    $observaciones   = $data['observaciones'] ?? null;
    
    if (!$idAnimal || !$litros) {
        http_response_code(400);
        echo json_encode(['error' => 'Campos requeridos: idAnimal, litros']);
        exit;
    }
    
    $sql = "INSERT INTO PRODUCCION_LECHE (id_animal, fecha_produccion, litros, turno, observaciones)
            VALUES (:id_animal, TO_DATE(:fecha, 'YYYY-MM-DD'), :litros, :turno, :observaciones)";
    $stmt = oci_parse($conn, $sql);
    oci_bind_by_name($stmt, ':id_animal',    $idAnimal);
    oci_bind_by_name($stmt, ':fecha',        $fechaProduccion);
    oci_bind_by_name($stmt, ':litros',       $litros);
    oci_bind_by_name($stmt, ':turno',        $turno);
    oci_bind_by_name($stmt, ':observaciones',$observaciones);
    
    if (!oci_execute($stmt, OCI_COMMIT_ON_SUCCESS)) {
        oci_free_statement($stmt);
        http_response_code(500);
        echo json_encode(['error' => 'Error al registrar producción']);
        exit;
    }
    oci_free_statement($stmt);
    http_response_code(201);
    echo json_encode(['success' => true, 'message' => 'Producción registrada'], JSON_UNESCAPED_UNICODE);
    exit;
}

// DELETE /api/produccion/{id} — Eliminar registro
if (preg_match('#^/api/produccion/(\d+)/?$#', $path, $matches) && $method === 'DELETE') {
    $id = $matches[1];
    $stmt = oci_parse($conn, "DELETE FROM PRODUCCION_LECHE WHERE id_produccion = :id");
    oci_bind_by_name($stmt, ':id', $id);
    oci_execute($stmt, OCI_COMMIT_ON_SUCCESS);
    oci_free_statement($stmt);
    http_response_code(200);
    echo json_encode(['success' => true, 'message' => 'Registro eliminado'], JSON_UNESCAPED_UNICODE);
    exit;
}

                FROM PESAJE p
                JOIN ANIMAL a ON p.id_animal = a.id_animal
                WHERE p.id_animal = :id
                ORDER BY p.fecha_pesaje DESC";
        $stmt = oci_parse($conn, $sql);
        oci_bind_by_name($stmt, ':id', $animalId);
    } else {
        $sql = "SELECT p.*, a.nombre as nombre_animal, a.codigo_identificacion as codigo
                FROM PESAJE p
                JOIN ANIMAL a ON p.id_animal = a.id_animal
                ORDER BY p.fecha_pesaje DESC";
        $stmt = oci_parse($conn, $sql);
    }
    
    oci_execute($stmt);
    $registros = [];
    while ($row = oci_fetch_assoc($stmt)) {
        $registros[] = [
            'id'            => (int)$row['ID_PESAJE'],
            'idAnimal'      => (int)$row['ID_ANIMAL'],
            'nombreAnimal'  => $row['NOMBRE_ANIMAL'],

            'pesoKg'        => (float)$row['PESO_KG'],
            'tipoPesaje'    => $row['TIPO_PESAJE'],
            'condicionCorp' => $row['CONDICION_CORP'] ? (float)$row['CONDICION_CORP'] : null,
            'observaciones' => $row['OBSERVACIONES'],
        ];
    }
    oci_free_statement($stmt);
    http_response_code(200);
    echo json_encode(['data' => $registros], JSON_UNESCAPED_UNICODE);
    exit;
}

// POST /api/pesaje
if (preg_match('#^/api/pesaje/?$#', $path) && $method === 'POST') {
    $idAnimal     = $data['idAnimal'] ?? null;
    $fechaPesaje  = $data['fechaPesaje'] ?? date('Y-m-d');
    $pesoKg       = $data['pesoKg'] ?? null;
    $tipoPesaje   = $data['tipoPesaje'] ?? 'periodico';
    $condicionCorp= $data['condicionCorp'] ?? null;
    $observaciones= $data['observaciones'] ?? null;
    
    if (!$idAnimal || !$pesoKg) {
        http_response_code(400);
        echo json_encode(['error' => 'Campos requeridos: idAnimal, pesoKg']);
        exit;
    }
    
    $sql = "INSERT INTO PESAJE (id_animal, fecha_pesaje, peso_kg, tipo_pesaje, condicion_corp, observaciones)
            VALUES (:id, TO_DATE(:fecha,'YYYY-MM-DD'), :peso, :tipo, :cond, :obs)";
    $stmt = oci_parse($conn, $sql);
    oci_bind_by_name($stmt, ':id',    $idAnimal);
    oci_bind_by_name($stmt, ':fecha', $fechaPesaje);
    oci_bind_by_name($stmt, ':peso',  $pesoKg);
    oci_bind_by_name($stmt, ':tipo',  $tipoPesaje);
    oci_bind_by_name($stmt, ':cond',  $condicionCorp);
    oci_bind_by_name($stmt, ':obs',   $observaciones);
    
    if (!oci_execute($stmt, OCI_COMMIT_ON_SUCCESS)) {
        http_response_code(500);
        echo json_encode(['error' => 'Error al registrar pesaje']);
        exit;
    }
    oci_free_statement($stmt);
    http_response_code(201);
    echo json_encode(['success' => true, 'message' => 'Pesaje registrado'], JSON_UNESCAPED_UNICODE);
    exit;
}

    $stmt = oci_parse($conn, "SELECT COUNT(*) as total FROM REPRODUCCION WHERE estado = 'gestacion'");
    oci_execute($stmt);
    $row = oci_fetch_assoc($stmt);
    $resumen['gestacionesActivas'] = (int)$row['TOTAL'];
    oci_free_statement($stmt);

    $stmt = oci_parse($conn, "SELECT COUNT(*) as total FROM SALUD
